📖 TF-IDF Vectorization

Python
from sklearn.feature_extraction.text import TfidfVectorizer

# Domain-specific text corpus
documents = [
    "property prices in Downtown reached new highs",
    "Metro Circle Line extension to improve connectivity",
    "Primary school enrollment increases in new estates",
    "Solar panels installed on building rooftops",
    "New market halls planned for Northpark and Newtown"
]

# TF-IDF: Term Frequency × Inverse Document Frequency
vectorizer = TfidfVectorizer(
    max_features=1000,      # Top 1000 words
    stop_words='english',   # Remove common words
    ngram_range=(1, 2)      # Include bigrams: "data pipeline", "model training"
)

X = vectorizer.fit_transform(documents)
print(f"Vocabulary size: {len(vectorizer.vocabulary_)}")
print(f"Feature matrix shape: {X.shape}")

# Top features per document
feature_names = vectorizer.get_feature_names_out()
for i, doc in enumerate(documents):
    scores = X[i].toarray().flatten()
    top_indices = scores.argsort()[-3:][::-1]
    top_words = [(feature_names[j], scores[j]) for j in top_indices]
    print(f"Doc {i}: {top_words}")

📖 Sentiment Analysis

Python
from sklearn.linear_model import LogisticRegression
from sklearn.pipeline import Pipeline

# Build a sentiment classifier
sentiment_pipeline = Pipeline([
    ('tfidf', TfidfVectorizer(max_features=5000, stop_words='english')),
    ('classifier', LogisticRegression(max_iter=1000))
])

# Training data (the region context)
train_texts = [
    "The new transit line is wonderful and efficient",        # positive
    "housing prices are unreasonably high for young couples", # negative
    "Clean and well-maintained market halls",            # positive
    "Terrible bus service with constant delays",           # negative
    "Beautiful parks and green spaces in Downtown",          # positive
    "Overcrowded trains during rush hour",                 # negative
]
train_labels = [1, 0, 1, 0, 1, 0]  # 1=positive, 0=negative

# Train
sentiment_pipeline.fit(train_texts, train_labels)

# Predict
new_texts = ["The renovation of Main Street is impressive"]
predictions = sentiment_pipeline.predict(new_texts)
probabilities = sentiment_pipeline.predict_proba(new_texts)
print(f"Sentiment: {'Positive' if predictions[0] == 1 else 'Negative'}")
print(f"Confidence: {max(probabilities[0]):.2%}")

📖 Text Preprocessing for Multi-Domain Text

Python
import re

def preprocess_sg_text(text):
    """Preprocess text with domain-specific considerations."""
    # Lowercase
    text = text.lower()

    # Handle Domain-specific abbreviations
    domain_abbrev = {
        'ml': 'machine learning',
        'nlp': 'natural language processing',
        'api': 'application programming interface',
        'etl': 'extract transform load',
        'eda': 'exploratory data analysis',
        'cv': 'computer vision',
        'dl': 'deep learning',
        'sql': 'structured query language',
    }
    for abbr, full in domain_abbrev.items():
        text = re.sub(r'\b' + abbr + r'\b', full, text)

    # Remove special characters but keep important ones
    text = re.sub(r'[^\w\s$%-]', '', text)

    # Remove extra whitespace
    text = re.sub(r'\s+', ' ', text).strip()

    return text

# Example
raw = "The API returned new ML models for the project"
print(preprocess_sg_text(raw))
# → "the application programming interface returned new machine learning models for the project"
⚠️ Common Mistakes
  • Not handling slang: "wow.*fancy.*competitive" won't be in standard stop word lists
  • Small vocabulary: max_features too low misses important domain terms
  • Ignoring class imbalance: If 90% of feedback is positive, the model just predicts "positive" always
