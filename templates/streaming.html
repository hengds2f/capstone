📖 Windowing Strategies

Concept
Time-based Windows for MRT Crowding Data:

Tumbling Window (fixed, non-overlapping):
|---5min---|---5min---|---5min---|
Event 1,2   Event 3,4   Event 5,6

Sliding Window (overlapping):
|---5min---|
   |---5min---|
      |---5min---|
Events may appear in multiple windows

Session Window (gap-based):
|--session 1--|  gap  |--session 2--|
Activity clusters separated by inactivity
Python
from datetime import datetime, timedelta
from collections import defaultdict

def tumbling_window(events, window_size_seconds=300):
    """Group events into fixed time windows."""
    windows = defaultdict(list)
    for event in events:
        ts = datetime.fromisoformat(event['timestamp'])
        # Floor to window boundary
        window_start = ts.replace(
            second=(ts.second // window_size_seconds) * window_size_seconds,
            microsecond=0
        )
        windows[window_start].append(event)
    return dict(windows)

# Process MRT crowding events
events = [
    {"station": "Bishan", "crowd_level": 0.8, "timestamp": "2024-01-15T08:01:00"},
    {"station": "Bishan", "crowd_level": 0.9, "timestamp": "2024-01-15T08:03:00"},
    {"station": "Orchard", "crowd_level": 0.7, "timestamp": "2024-01-15T08:02:00"},
]

windows = tumbling_window(events)
for window_start, window_events in windows.items():
    avg_crowd = sum(e['crowd_level'] for e in window_events) / len(window_events)
    print(f"Window {window_start}: {len(window_events)} events, avg crowd: {avg_crowd:.2f}")

# Real-time anomaly detection using z-score
import statistics

class StreamAnomalyDetector:
    def __init__(self, window_size=50, threshold=2.0):
        self.window = []
        self.window_size = window_size
        self.threshold = threshold
        self.anomalies = []

    def process(self, value, metadata=None):
        self.window.append(value)
        if len(self.window) > self.window_size:
            self.window.pop(0)

        if len(self.window) >= 10:  # Need minimum data
            mean = statistics.mean(self.window)
            stdev = statistics.stdev(self.window)
            if stdev > 0:
                z_score = abs(value - mean) / stdev
                if z_score > self.threshold:
                    self.anomalies.append({
                        'value': value, 'z_score': z_score,
                        'mean': mean, 'metadata': metadata
                    })
                    return True  # Anomaly detected!
        return False

# Usage: Monitor MRT crowding
detector = StreamAnomalyDetector(threshold=2.5)
for event in mrt_events:
    is_anomaly = detector.process(event['crowd_level'], event['station'])
    if is_anomaly:
        print(f"🚨 ALERT: Unusual crowding at {event['station']}!")
