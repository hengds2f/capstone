📋 EDA Checklist

Python
import pandas as pd
import numpy as np

df = pd.read_csv('data/hdb_resale.csv')

# 1. Shape and structure
print(f"Shape: {df.shape}")
print(f"Columns: {list(df.columns)}")
print(f"Dtypes:\n{df.dtypes}")

# 2. Missing data analysis
print(f"\nNull counts:\n{df.isnull().sum()}")
print(f"Null %:\n{(df.isnull().sum() / len(df) * 100).round(2)}")

# 3. Statistical summary
print(f"\nDescribe:\n{df.describe()}")

# 4. Distribution analysis
print(f"\nUnique towns: {df['town'].nunique()}")
print(f"Value counts:\n{df['flat_type'].value_counts()}")

# 5. Correlation analysis
numeric = df.select_dtypes(include=[np.number])
print(f"\nCorrelation matrix:\n{numeric.corr().round(3)}")

# 6. Outlier detection (IQR method)
Q1 = df['resale_price'].quantile(0.25)
Q3 = df['resale_price'].quantile(0.75)
IQR = Q3 - Q1
outliers = df[(df['resale_price'] < Q1 - 1.5*IQR) | (df['resale_price'] > Q3 + 1.5*IQR)]
print(f"\nOutliers found: {len(outliers)}")
