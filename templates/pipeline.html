📖 Pipeline Patterns

Retry with Exponential Backoff

Python
import time, random

def retry_with_backoff(func, max_retries=3, base_delay=1):
    """Retry a function with exponential backoff + jitter."""
    for attempt in range(max_retries):
        try:
            return func()
        except Exception as e:
            if attempt == max_retries - 1:
                raise
            delay = base_delay * (2 ** attempt) + random.uniform(0, 1)
            print(f"Attempt {attempt+1} failed: {e}. Retrying in {delay:.1f}s")
            time.sleep(delay)

# Usage
retry_with_backoff(lambda: ingest_csv_to_db("hdb_resale.csv"))

Idempotent Pipeline Steps

Python
# BAD: Appends duplicates on re-run
def ingest_bad(df, table):
    df.to_sql(table, conn, if_exists='append')

# GOOD: Idempotent — safe to re-run
def ingest_good(df, table):
    conn.execute(f"DELETE FROM {table}")  # Clear first
    df.to_sql(table, conn, if_exists='append')

# BETTER: Upsert pattern
def ingest_upsert(df, table, key_col):
    for _, row in df.iterrows():
        conn.execute(f"""
            INSERT INTO {table} ({', '.join(row.index)})
            VALUES ({', '.join(['?']*len(row))})
            ON CONFLICT({key_col}) DO UPDATE SET
            {', '.join(f'{c}=excluded.{c}' for c in row.index if c != key_col)}
        """, tuple(row))
