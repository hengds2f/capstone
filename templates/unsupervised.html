Cluster neighborhoods by housing characteristics (average price, area, transaction volume).

📖 K-Means Algorithm

Python
from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler
import pandas as pd

# Prepare neighborhood features
neighborhood_data = df.groupby('town').agg(
    avg_price=('price', 'mean'),
    avg_area=('area', 'mean'),
    transaction_count=('price', 'count'),
    price_std=('price', 'std')
).reset_index()

# Always scale features before clustering!
scaler = StandardScaler()
features = scaler.fit_transform(neighborhood_data[['avg_price', 'avg_area', 'transaction_count']])

# Fit K-Means
kmeans = KMeans(n_clusters=3, random_state=42, n_init=10)
neighborhood_data['cluster'] = kmeans.fit_predict(features)

# Interpret clusters
for cluster_id in range(3):
    cluster = neighborhood_data[neighborhood_data['cluster'] == cluster_id]
    print(f"\nCluster {cluster_id}: {list(cluster['town'])}")
    print(f"  Avg price: ${cluster['avg_price'].mean():,.0f}")
    print(f"  Avg area: {cluster['avg_area'].mean():.0f} sqm")

Choosing K: The Elbow Method

Python
# Try different values of K and plot inertia (within-cluster sum of squares)
inertias = []
K_range = range(1, 10)
for k in K_range:
    km = KMeans(n_clusters=k, random_state=42, n_init=10)
    km.fit(features)
    inertias.append(km.inertia_)

# Plot elbow curve
import plotly.graph_objects as go
fig = go.Figure(data=go.Scatter(x=list(K_range), y=inertias, mode='lines+markers'))
fig.update_layout(title='Elbow Method', xaxis_title='K', yaxis_title='Inertia')
fig.show()

# Look for the "elbow" — where adding more clusters stops reducing inertia significantly
# Silhouette score confirms optimal K
from sklearn.metrics import silhouette_score
for k in range(2, 8):
    km = KMeans(n_clusters=k, random_state=42, n_init=10)
    labels = km.fit_predict(features)
    score = silhouette_score(features, labels)
    print(f"K={k}: Silhouette Score = {score:.3f}")

📖 PCA — Dimensionality Reduction

Python
from sklearn.decomposition import PCA

# Reduce 5 features to 2 dimensions for visualization
pca = PCA(n_components=2)
reduced = pca.fit_transform(features)

print(f"Explained variance: {pca.explained_variance_ratio_}")
# e.g., [0.65, 0.25] — first 2 components explain 90% of variation

# Plot clusters in 2D
fig = go.Figure()
for cluster_id in range(3):
    mask = neighborhood_data['cluster'] == cluster_id
    fig.add_trace(go.Scatter(
        x=reduced[mask, 0], y=reduced[mask, 1],
        mode='markers+text',
        text=neighborhood_data[mask]['town'],
        name=f'Cluster {cluster_id}'
    ))
fig.update_layout(title='Neighborhoods — PCA Visualization')
fig.show()
⚠️ Common Mistakes
  • Not scaling features: K-Means uses distance — unscaled features with large ranges dominate
  • Choosing K arbitrarily: Use elbow method + silhouette score + domain knowledge
  • Interpreting clusters without context: Always map clusters back to meaningful labels
