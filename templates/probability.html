📖 Descriptive Statistics

Python
import numpy as np
import pandas as pd

# HDB resale prices for Bishan
prices = np.array([480000, 520000, 510000, 650000, 490000, 530000, 1200000, 500000])

# Measures of Central Tendency
mean = np.mean(prices)          # 610,000 — pulled up by outlier
median = np.median(prices)      # 515,000 — robust to outliers
# Mode: most frequent value

# Measures of Spread
std = np.std(prices)           # Standard deviation
variance = np.var(prices)      # Variance = std²
iqr = np.percentile(prices, 75) - np.percentile(prices, 25)  # IQR

# Skewness: positive = right tail (common in prices)
from scipy.stats import skew
price_skew = skew(prices)      # > 0 means right-skewed

print(f"Mean: ${mean:,.0f}, Median: ${median:,.0f}")
print(f"Std Dev: ${std:,.0f}, IQR: ${iqr:,.0f}")
print(f"Skewness: {price_skew:.2f} (right-skewed = expensive outliers)")

Python
from scipy import stats

# Poisson: MRT incidents per month (avg = 2.5)
poisson_dist = stats.poisson(mu=2.5)
print(f"P(0 incidents) = {poisson_dist.pmf(0):.4f}")
print(f"P(≤3 incidents) = {poisson_dist.cdf(3):.4f}")

# Normal: HDB prices ~ N(500000, 80000²)
normal_dist = stats.norm(loc=500000, scale=80000)
print(f"P(price < 400000) = {normal_dist.cdf(400000):.4f}")
print(f"P(400000 < price < 600000) = {normal_dist.cdf(600000) - normal_dist.cdf(400000):.4f}")

# Central Limit Theorem: sample means are normally distributed
sample_means = [np.mean(np.random.choice(prices, size=10)) for _ in range(1000)]
print(f"Sample mean distribution: μ={np.mean(sample_means):,.0f}, σ={np.std(sample_means):,.0f}")

📖 Hypothesis Testing

Python
# Question: Are Bishan HDB prices significantly higher than Tampines?
bishan_prices = [520000, 530000, 540000, 550000, 560000, 580000]
tampines_prices = [420000, 430000, 440000, 450000, 460000, 470000]

# H₀: μ_bishan = μ_tampines (no difference)
# H₁: μ_bishan > μ_tampines (Bishan is higher)

t_stat, p_value = stats.ttest_ind(bishan_prices, tampines_prices)
alpha = 0.05

print(f"t-statistic: {t_stat:.4f}")
print(f"p-value: {p_value:.6f}")
print(f"Reject H₀? {'Yes' if p_value < alpha else 'No'} (α={alpha})")
# Result: Yes — Bishan prices are statistically significantly higher

📖 Correlation & Causation

# Pearson correlation: linear relationship between two variables
# Spearman: monotonic relationship (robust to outliers)

# floor_area_sqm vs resale_price

# ⚠️ Correlation ≠ Causation
# Observation: Ice cream sales ↑ and dengue cases ↑ in Singapore
# Correlation: r = 0.7
# Causation? NO — both are caused by hot weather (confounding variable)
⚠️ Common Mistakes
  • Using mean for skewed data: HDB prices are right-skewed — use median instead
  • p-value ≠ probability H₀ is true: It's P(data | H₀), not P(H₀ | data)
  • Multiple testing problem: Testing 20 hypotheses at α=0.05 ⇒ expect 1 false positive
