📖 Why SQL for Data Science?

SQL (Structured Query Language) is the universal language for working with relational databases. In Singapore's data ecosystem, government datasets from HDB, URA, and LTA are often stored in relational databases. SQL lets you query, filter, aggregate, and join these datasets efficiently.

1️⃣ SQL DDL — Data Definition Language

DDL commands define the structure of your database: tables, columns, constraints, and indexes.

🎯 Purpose

Design a normalized schema for Singapore housing data with proper relationships and constraints.

SQL
-- CREATE TABLE with constraints
CREATE TABLE dim_location (
    location_id INTEGER PRIMARY KEY AUTOINCREMENT,
    town VARCHAR(100) NOT NULL,
    flat_type VARCHAR(50),
    block VARCHAR(20),
    street_name VARCHAR(200),
    planning_area VARCHAR(100),
    UNIQUE(town, block, street_name, flat_type)
);

-- CREATE INDEX for query performance
CREATE INDEX idx_location_town ON dim_location(town);

-- ALTER TABLE (in SQLite, limited to adding columns)
-- ALTER TABLE dim_location ADD COLUMN postal_code VARCHAR(10);

-- CREATE VIEW for reusable queries
CREATE VIEW v_hdb_summary AS
SELECT l.town, t.year, AVG(f.resale_price) as avg_price
FROM fact_hdb_transactions f
JOIN dim_location l ON f.location_id = l.location_id
JOIN dim_time t ON f.time_id = t.time_id
GROUP BY l.town, t.year;
Expected: Tables created with proper constraints and relationships

2️⃣ SQL DML — Data Manipulation Language

DML commands insert, update, delete, and query data.

INSERT

SQL
-- Single INSERT
INSERT INTO raw_hdb_resale (month, town, flat_type, resale_price)
VALUES ('2024-07', 'BISHAN', '4 ROOM', 650000);

-- Bulk INSERT
INSERT INTO dim_time (full_date, year, quarter, month, month_name)
VALUES
    ('2024-01-01', 2024, 1, 1, 'January'),
    ('2024-02-01', 2024, 1, 2, 'February'),
    ('2024-03-01', 2024, 1, 3, 'March');

SELECT with Filters

SQL
SELECT town, flat_type, resale_price
FROM raw_hdb_resale
WHERE resale_price > 500000
ORDER BY resale_price DESC;

UPDATE and DELETE

SQL
-- UPDATE (be careful in production!)
UPDATE raw_hdb_resale
SET resale_price = 660000
WHERE town = 'BISHAN' AND block = '184' AND month = '2024-01';

-- DELETE with condition
DELETE FROM raw_hdb_resale WHERE resale_price < 0;

-- TRANSACTION for data integrity
BEGIN TRANSACTION;
    INSERT INTO raw_hdb_resale (month, town, flat_type, resale_price)
    VALUES ('2024-07', 'PUNGGOL', '4 ROOM', 550000);
    UPDATE pipeline_runs SET status = 'completed' WHERE run_id = 1;
COMMIT;

3️⃣ Advanced SQL

JOINs

SQL
-- Multi-table JOIN: Combine facts with dimensions
SELECT l.town, t.year, t.month, p.flat_type,
       f.resale_price, f.price_per_sqm
FROM fact_hdb_transactions f
JOIN dim_location l ON f.location_id = l.location_id
JOIN dim_time t ON f.time_id = t.time_id
JOIN dim_property_type p ON f.property_type_id = p.property_type_id
ORDER BY f.resale_price DESC LIMIT 15;

GROUP BY, HAVING

SQL
SELECT town, COUNT(*) as transactions,
       ROUND(AVG(resale_price), 0) as avg_price,
       MIN(resale_price) as min_price,
       MAX(resale_price) as max_price
FROM raw_hdb_resale
GROUP BY town
HAVING COUNT(*) >= 2
ORDER BY avg_price DESC;

Subqueries

SQL
-- Towns with above-average prices
SELECT town, resale_price, flat_type
FROM raw_hdb_resale
WHERE resale_price > (SELECT AVG(resale_price) FROM raw_hdb_resale)
ORDER BY resale_price DESC;

CTEs (Common Table Expressions)

SQL
WITH town_stats AS (
    SELECT town, AVG(resale_price) as avg_price, COUNT(*) as cnt
    FROM raw_hdb_resale GROUP BY town
),
overall AS (
    SELECT AVG(resale_price) as overall_avg FROM raw_hdb_resale
)
SELECT t.town, ROUND(t.avg_price, 0) as avg_price, t.cnt,
       ROUND(o.overall_avg, 0) as overall_avg,
       ROUND(t.avg_price - o.overall_avg, 0) as diff
FROM town_stats t, overall o
ORDER BY diff DESC;

Window Functions

SQL
SELECT town, flat_type, resale_price,
    RANK() OVER (PARTITION BY town ORDER BY resale_price DESC) as price_rank,
    ROUND(AVG(resale_price) OVER (PARTITION BY town), 0) as town_avg
FROM raw_hdb_resale
ORDER BY town, price_rank;
⚠️ Common SQL Mistakes
  • SELECT * in production: Always specify columns to avoid unnecessary data transfer
  • Missing WHERE on UPDATE/DELETE: Can corrupt your entire table
  • Not using indexes: Queries on large tables become extremely slow
  • Cartesian joins: Forgetting JOIN conditions creates massive result sets
  • NULL handling: NULL is not equal to anything — use IS NULL, not = NULL
